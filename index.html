        function getCurrentWeekActivities() {

            const today = new Date();

            const currentWeek = activities.find(week => today >= week.start && today <= week.end);

            return currentWeek ? currentWeek : { week: 'No activities found for this week.', items: [] };

        }


        function displayActivities() {

            const currentWeek = getCurrentWeekActivities();

            activitiesDiv.innerHTML = `

                <h2>${currentWeek.week}</h2>

                <ul>

                    ${currentWeek.items.map(item => `<li>${item}</li>`).join('')}

                </ul>

            `;
